-- ===========================================================
-- USERS & ACCESS CONTROL
-- ===========================================================
CREATE TABLE ACL (
    RoleID CHAR(10) PRIMARY KEY,
    RoleName NVARCHAR(100),
    Permissions NVARCHAR(MAX) -- JSON or comma-separated list
);

CREATE TABLE Users (
    UserID CHAR(10) PRIMARY KEY,
    FullName NVARCHAR(150),
    Email NVARCHAR(150) UNIQUE,
    PasswordHash NVARCHAR(256),
    RoleID CHAR(10) FOREIGN KEY REFERENCES ACL(RoleID),
    IsActive BIT DEFAULT 1,
    CreatedAt DATETIME DEFAULT GETDATE()
);

CREATE TABLE UserSessions (
    SessionID CHAR(10) PRIMARY KEY,
    UserID CHAR(10) FOREIGN KEY REFERENCES Users(UserID),
    LoginTime DATETIME DEFAULT GETDATE(),
    LogoutTime DATETIME NULL,
    IPAddress NVARCHAR(50),
    DeviceInfo NVARCHAR(150)
);

-- ===========================================================
-- REGISTRATION & CUSTOMERS
-- ===========================================================
CREATE TABLE Registration (
    RegID NVARCHAR(10) PRIMARY KEY,  -- Note: Using NVARCHAR for consistency
    FullName NVARCHAR(150),
    CNIC NVARCHAR(50),
    Phone NVARCHAR(50),
    Email NVARCHAR(150),
    CreatedAt DATETIME DEFAULT GETDATE(),
    Status NVARCHAR(50) DEFAULT 'Pending'
);
-- PaymentPlan is Parent
CREATE TABLE PaymentPlan (
    PlanID CHAR(10) PRIMARY KEY,
    ProjectID CHAR(10) NULL, -- may be assigned later
    PlanName NVARCHAR(150),
    TotalAmount DECIMAL(18,2),
    DurationMonths INT,
    Frequency NVARCHAR(50), -- e.g. Monthly, Quarterly
    Description NVARCHAR(255),
    CreatedAt DATETIME DEFAULT GETDATE()
);
--PaymentSchedule is child of PaymentPlan
CREATE TABLE PaymentSchedule (
    ScheduleID CHAR(10) PRIMARY KEY,
    PlanID CHAR(10) FOREIGN KEY REFERENCES PaymentPlan(PlanID),
    PaymentDescription nvarchar(250),
    InstallmentNo int,

    DueDate DATE,
    Amount DECIMAL(18,2),
    SurchargeApplied BIT DEFAULT 1,
    SurchargeRate DECIMAL(18,2) DEFAULT 0.05,
    Description NVARCHAR(255)
);

CREATE TABLE Customers (
    CustomerID CHAR(10) PRIMARY KEY,
    RegID NVARCHAR(10) NULL FOREIGN KEY REFERENCES Registration(RegID) ON DELETE SET NULL, -- NULLABLE: Customer can be created with or without registration
    PlanID CHAR(10) FOREIGN KEY REFERENCES PaymentPlan(PlanID),

    -- Personal Info
    FullName NVARCHAR(150),
    FatherName NVARCHAR(150),
    CNIC NVARCHAR(50),
    PassportNo NVARCHAR(50),
    DOB DATE,
    Gender NVARCHAR(20),

    -- Contact Info
    Phone NVARCHAR(50),
    Email NVARCHAR(150),
    MailingAddress NVARCHAR(255),
    PermanentAddress NVARCHAR(255),
    City NVARCHAR(100),
    Country NVARCHAR(100),

    -- Project Info
    SubProject NVARCHAR(100),
    RegisteredSize NVARCHAR(50),

    -- System Info
    CreatedAt DATETIME DEFAULT GETDATE(),
    Status NVARCHAR(50) DEFAULT 'Active',


     NomineeName NVARCHAR(100),
    NomineeID NVARCHAR(50),
     NomineeRelation NVARCHAR(50),
     AdditionalInfo NVARCHAR(max)
);

CREATE TABLE CustomerLogs (
    LogID INT IDENTITY PRIMARY KEY,
    CustomerID CHAR(10) FOREIGN KEY REFERENCES Customers(CustomerID),
    Action NVARCHAR(150),
    Remarks NVARCHAR(255),
    CreatedAt DATETIME DEFAULT GETDATE(),
    CreatedBy CHAR(10)
);

-- ===========================================================
-- PROJECTS & PROPERTIES
-- ===========================================================
CREATE TABLE Projects (
    ProjectID CHAR(10) PRIMARY KEY,
    ProjectName NVARCHAR(150),
    Type NVARCHAR(50),
    Location NVARCHAR(150),
    Description NVARCHAR(MAX),
    CreatedAt DATETIME DEFAULT GETDATE()
);

CREATE TABLE Property (
    PropertyID CHAR(10) PRIMARY KEY,
    ProjectID CHAR(10) FOREIGN KEY REFERENCES Projects(ProjectID),
   
    PlotNo NVARCHAR(50),
    Street NVARCHAR(50),
    PlotType NVARCHAR(50),
    [Block] NVARCHAR(50),
    PropertyType NVARCHAR(50),
    Size NVARCHAR(50),
    Status NVARCHAR(50) DEFAULT 'Available',
    CreatedAt DATETIME DEFAULT GETDATE(),
    AdditionalInfo NVARCHAR(max)
);

CREATE TABLE Allotment (
    AllotmentID CHAR(10) PRIMARY KEY,
    PropertyID CHAR(10) FOREIGN KEY REFERENCES Property(PropertyID),
    CustomerID CHAR(10) FOREIGN KEY REFERENCES Customers(CustomerID),
    AllottedBy CHAR(10) FOREIGN KEY REFERENCES Users(UserID),
    AllotmentDate DATETIME DEFAULT GETDATE(),
    ApprovedBy NVARCHAR(50),
    AllottmentType NVARCHAR(50),
    WorkFlowStatus NVARCHAR(250),
    Comments NVARCHAR(max),
    AdditionalInfo NVARCHAR(max)
);

CREATE TABLE Balloting (
    BallotID CHAR(10) PRIMARY KEY,
    ProjectID CHAR(10) FOREIGN KEY REFERENCES Projects(ProjectID),
    ConductedBy CHAR(10) FOREIGN KEY REFERENCES Users(UserID),
    ConductedAt DATETIME DEFAULT GETDATE(),
    Remarks NVARCHAR(255)
);

CREATE TABLE Possession (
    PossessionID CHAR(10) PRIMARY KEY,
    PropertyID CHAR(10) FOREIGN KEY REFERENCES Property(PropertyID),
    CustomerID CHAR(10) FOREIGN KEY REFERENCES Customers(CustomerID),
    PossessionDate DATETIME DEFAULT GETDATE(),
      WorkFlowStatus NVARCHAR(250),
       Comments NVARCHAR(max),
    Remarks NVARCHAR(255)
);

-- ===========================================================
-- PAYMENT FLOW
-- ===========================================================




CREATE TABLE Payments (
    PaymentID CHAR(10) PRIMARY KEY,
    ScheduleID CHAR(10) FOREIGN KEY REFERENCES PaymentSchedule(ScheduleID),
    CustomerID NVARCHAR(10) FOREIGN KEY REFERENCES Customers(CustomerID),
    PaymentDate DATETIME DEFAULT GETDATE(),
    Amount DECIMAL(18,2),
    Method NVARCHAR(50),
    ReferenceNo NVARCHAR(100),
    [Status] NVARCHAR(250),
    Remarks NVARCHAR(255)
);

CREATE TABLE Penalties (
    PenaltyID CHAR(10) PRIMARY KEY,
    CustomerID CHAR(10) FOREIGN KEY REFERENCES Customers(CustomerID),
    Amount DECIMAL(18,2),
    Reason NVARCHAR(255),
    AppliedOn DATETIME DEFAULT GETDATE()
);

CREATE TABLE Waiver (
    WaiverID CHAR(10) PRIMARY KEY,
    CustomerID CHAR(10) FOREIGN KEY REFERENCES Customers(CustomerID),
    Amount DECIMAL(18,2),
    Reason NVARCHAR(255),
    ApprovedBy CHAR(10) FOREIGN KEY REFERENCES Users(UserID),
    CreatedAt DATETIME DEFAULT GETDATE()
);

CREATE TABLE Refund (
    RefundID CHAR(10) PRIMARY KEY,
    CustomerID CHAR(10) FOREIGN KEY REFERENCES Customers(CustomerID),
    Amount DECIMAL(18,2),
    Reason NVARCHAR(255),
    Status NVARCHAR(50),
    ApprovedBy CHAR(10) FOREIGN KEY REFERENCES Users(UserID),
    CreatedAt DATETIME DEFAULT GETDATE()
);

-- ===========================================================
-- TRANSFERS, NDC, APPROVALS
-- ===========================================================
CREATE TABLE Transfer (
    TransferID CHAR(10) PRIMARY KEY,
    FromCustomerID CHAR(10) FOREIGN KEY REFERENCES Customers(CustomerID),
    ToCustomerID CHAR(10) FOREIGN KEY REFERENCES Customers(CustomerID),
    PropertyID CHAR(10) FOREIGN KEY REFERENCES Property(PropertyID),
    Status NVARCHAR(50),
    CreatedAt DATETIME DEFAULT GETDATE()
);

CREATE TABLE NDC (
    NDCID CHAR(10) PRIMARY KEY,
    CustomerID CHAR(10) FOREIGN KEY REFERENCES Customers(CustomerID),
    NDCType nvarchar(100),
    Title nvarchar(500),
    WorkFlowStatus nvarchar(500),
    Comments  nvarchar(max),
    IssuedDate DATETIME DEFAULT GETDATE(),
    Remarks NVARCHAR(255)
);

CREATE TABLE Approvals (
    ApprovalID CHAR(10) PRIMARY KEY,
    RefType NVARCHAR(50),
    RefID CHAR(10),
    ApprovedBy NVARCHAR(max),
    Status NVARCHAR(50),
    Remarks NVARCHAR(255),
    CreatedAt DATETIME DEFAULT GETDATE()
);

-- ===========================================================
-- ATTACHMENTS, ACTIVITY LOGS & NOTIFICATIONS
-- ===========================================================
CREATE TABLE Attachments (
    AttachmentID CHAR(10) PRIMARY KEY,
    RefType NVARCHAR(50),
    RefID CHAR(10),
    FilePath NVARCHAR(255),
    UploadedBy CHAR(10) FOREIGN KEY REFERENCES Users(UserID),
    UploadedAt DATETIME DEFAULT GETDATE()
);

CREATE TABLE ActivityLog (
    LogID INT IDENTITY PRIMARY KEY,
    UserID CHAR(10) FOREIGN KEY REFERENCES Users(UserID),
    Action NVARCHAR(255),
    RefType NVARCHAR(50),
    RefID CHAR(10),
    CreatedAt DATETIME DEFAULT GETDATE()
);

CREATE TABLE Notifications (
    NotificationID CHAR(10) PRIMARY KEY,
    UserID CHAR(10) FOREIGN KEY REFERENCES Users(UserID),
    Message NVARCHAR(255),
    IsRead BIT DEFAULT 0,
    CreatedAt DATETIME DEFAULT GETDATE()
);

-- ===========================================================
-- CONFIGURATION & SETTINGS
-- ===========================================================
CREATE TABLE Configuration (
    ConfigKey NVARCHAR(50) PRIMARY KEY,
    Category NVARCHAR(50),
    ConfigValue NVARCHAR(MAX),
    Description NVARCHAR(255),
    CreatedAt DATETIME DEFAULT GETDATE(),
    UpdatedAt DATETIME NULL,
    UpdatedBy CHAR(10) NULL
);
GO
