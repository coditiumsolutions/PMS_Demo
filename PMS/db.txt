-- ===========================================================
-- USERS & ACCESS CONTROL
-- ===========================================================
CREATE TABLE ACL (
    RoleID CHAR(10) PRIMARY KEY,
    RoleName NVARCHAR(100),
    Permissions NVARCHAR(MAX) -- JSON or comma-separated list
);

CREATE TABLE Users (
    UserID CHAR(10) PRIMARY KEY,
    FullName NVARCHAR(150),
    Email NVARCHAR(150) UNIQUE,
    PasswordHash NVARCHAR(256),
    RoleID CHAR(10) FOREIGN KEY REFERENCES ACL(RoleID),
    Designation NVARCHAR(150) NULL,
    Department NVARCHAR(150) NULL,
    UserType NVARCHAR(50) NULL,  -- Admin / Manager / CRO
    IsActive BIT DEFAULT 1,
    CreatedAt DATETIME DEFAULT GETDATE()
);

-- Per-user, per-module access: NoAccess | Read | Edit | Admin (NoAccess=no access, Read=view only, Edit=Create+Edit+Read, Admin=full CRUD)
CREATE TABLE UserModulePermission (
    UserID CHAR(10) NOT NULL FOREIGN KEY REFERENCES Users(UserID) ON DELETE CASCADE,
    ModuleKey NVARCHAR(50) NOT NULL,
    Permission NVARCHAR(20) NOT NULL,  -- NoAccess, Read, Edit, Admin
    PRIMARY KEY (UserID, ModuleKey)
);

CREATE TABLE Tickets (
    TicketID CHAR(10) PRIMARY KEY,
    CustomerID NVARCHAR(150),
    Email NVARCHAR(150) UNIQUE,
    Contact NVARCHAR(256),
    Title NVARCHAR(max),
    Description NVARCHAR(max),
    CROComments NVARCHAR(max),
    Status NVARCHAR(256),
    CreatedBy NVARCHAR(256),
    AssignedTo NVARCHAR(256),
    TicketClosingDate DATETIME ,
    CreatedAt DATETIME DEFAULT GETDATE()
);

CREATE TABLE UserSessions (
    SessionID CHAR(10) PRIMARY KEY,
    UserID CHAR(10) FOREIGN KEY REFERENCES Users(UserID),
    LoginTime DATETIME DEFAULT GETDATE(),
    LogoutTime DATETIME NULL,
    IPAddress NVARCHAR(50),
    DeviceInfo NVARCHAR(150)
);

-- ===========================================================
-- REGISTRATION & CUSTOMERS
-- ===========================================================
CREATE TABLE Registration (
    RegID NVARCHAR(10) PRIMARY KEY,  -- Note: Using NVARCHAR for consistency
    FullName NVARCHAR(150),
    CNIC NVARCHAR(50),               -- Strict format: XXXXX-XXXXXXX-X (digits only)
    PassportNo NVARCHAR(50) NULL,    -- Alphanumeric, required if no CNIC
    Phone NVARCHAR(50),
    Email NVARCHAR(150),
    ProjectID CHAR(10) NULL,          -- FK to Projects (optional)
    Size NVARCHAR(100) NULL,         -- from selected project's Sizes (CSV)
    CreatedAt DATETIME DEFAULT GETDATE(),
    Status NVARCHAR(50) DEFAULT 'Pending',
    CONSTRAINT FK_Registration_Project FOREIGN KEY (ProjectID) REFERENCES Projects(ProjectID) ON DELETE SET NULL
);
-- PaymentPlan is Parent
CREATE TABLE PaymentPlan (
    PlanID CHAR(10) PRIMARY KEY,
    ProjectID CHAR(10) NULL, -- may be assigned later
    PlanName NVARCHAR(150),
    TotalAmount DECIMAL(18,2),
    TotalAmountUSD DECIMAL(18,2),
    DurationMonths INT,
    Frequency NVARCHAR(50), -- e.g. Monthly, Quarterly
    Currency NVARCHAR(10) NOT NULL DEFAULT 'PKR',
    ExchangeRate DECIMAL(18,2),
    Description NVARCHAR(255),
    CreatedAt DATETIME DEFAULT GETDATE()
);
--PaymentSchedule is child of PaymentPlan
CREATE TABLE PaymentSchedule (
    ScheduleID CHAR(10) PRIMARY KEY,
    PlanID CHAR(10) FOREIGN KEY REFERENCES PaymentPlan(PlanID),
    PaymentDescription nvarchar(250),
    InstallmentNo int,

    DueDate DATE,
    Amount DECIMAL(18,2),
    SurchargeApplied BIT DEFAULT 1,
    SurchargeRate DECIMAL(18,2) DEFAULT 0.05,
    Description NVARCHAR(255)
);

CREATE TABLE Customers (
    CustomerID CHAR(10) PRIMARY KEY,
    RegID NVARCHAR(10) NULL FOREIGN KEY REFERENCES Registration(RegID) ON DELETE SET NULL, -- NULLABLE: Customer can be created with or without registration
    PlanID CHAR(10) FOREIGN KEY REFERENCES PaymentPlan(PlanID),
    ProjectID NVARCHAR(10) NULL, -- Project assignment

    -- Personal Info
    FullName NVARCHAR(150),
    FatherName NVARCHAR(150),
    CNIC NVARCHAR(50),
    PassportNo NVARCHAR(50),
    DOB DATE,
    Gender NVARCHAR(20),
    Nationality NVARCHAR(100),

    -- Contact Info
    Phone NVARCHAR(50),
    Email NVARCHAR(150),
    MailingAddress NVARCHAR(255),
    PermanentAddress NVARCHAR(255),
    City NVARCHAR(100),
    Country NVARCHAR(100),

    -- Project Info
    SubProject NVARCHAR(100),
    RegisteredSize NVARCHAR(50),

    -- System Info
    CreatedAt DATETIME DEFAULT GETDATE(),
    Status NVARCHAR(50) DEFAULT 'Active',

    -- Nominee Info
    NomineeName NVARCHAR(100),
    NomineeID NVARCHAR(50),
    NomineeRelation NVARCHAR(50),
    NomineeNICDocumentPath NVARCHAR(255),
    NomineePicturePath NVARCHAR(255),
    AdditionalInfo NVARCHAR(max),
    DealerID int NULL FOREIGN KEY REFERENCES Dealers(DealerID),
    isDealerRegistered int NULL,   -- 1=Yes (use DealerID), 0=No (use DealerName)
    DealerName NVARCHAR(200) NULL -- Manual dealer name when isDealerRegistered=0
);

CREATE TABLE CustomerLogs (
    LogID INT IDENTITY PRIMARY KEY,
    CustomerID CHAR(10) FOREIGN KEY REFERENCES Customers(CustomerID),
    Action NVARCHAR(150),
    Remarks NVARCHAR(255),

-- ===========================================================

CREATE TABLE Property (
    PropertyID CHAR(10) PRIMARY KEY,
    ProjectID CHAR(10) FOREIGN KEY REFERENCES Projects(ProjectID),
   
    PlotNo NVARCHAR(50),
    Street NVARCHAR(50),
    PlotType NVARCHAR(50),
    [Block] NVARCHAR(50),
    Floor NVARCHAR(50),

    Status NVARCHAR(50) DEFAULT 'Available',
    CreatedAt DATETIME DEFAULT GETDATE(),
    AdditionalInfo NVARCHAR(max),
    DealerID int
);

-- ===========================================================
-- RENTAL MODULE (Organization-owned inventory rentals)
-- Use case: PMS Organization rents out an available property (NOT sold/allotted customer-owned).
-- ===========================================================
CREATE TABLE Rental (
    RentalID NVARCHAR(50) PRIMARY KEY,
    PropertyID CHAR(10) NOT NULL FOREIGN KEY REFERENCES Property(PropertyID),

    -- Tenant info (stored here, NOT linked to Customers)
    TenantName NVARCHAR(200) NOT NULL,
    TenantCNIC NVARCHAR(50) NULL,
    TenantPhone NVARCHAR(50) NULL,
    TenantEmail NVARCHAR(150) NULL,
    TenantAddress NVARCHAR(255) NULL,

    -- Rental terms
    MonthlyRent DECIMAL(18,2) NOT NULL,
    SecurityDeposit DECIMAL(18,2) NULL,
    AdvanceRent DECIMAL(18,2) NULL,
    Currency NVARCHAR(10) NOT NULL DEFAULT 'PKR',
    StartDate DATE NOT NULL,
    EndDate DATE NULL,
    DurationMonths INT NOT NULL,
    PaymentDueDayOfMonth INT NULL, -- recommended 1-28
    Status NVARCHAR(50) NOT NULL DEFAULT 'Active', -- Active, Completed, Cancelled
    Notes NVARCHAR(MAX) NULL,
    CreatedAt DATETIME NOT NULL DEFAULT GETDATE(),
    UpdatedAt DATETIME NULL
);

-- One active rental per property (industry standard)
CREATE UNIQUE INDEX UX_Rental_Property_Active
ON Rental(PropertyID)
WHERE Status = 'Active';

CREATE TABLE RentalPayments (
    RentalPaymentID NVARCHAR(50) PRIMARY KEY,
    RentalID NVARCHAR(50) NOT NULL FOREIGN KEY REFERENCES Rental(RentalID) ON DELETE CASCADE,

    DueDate DATE NOT NULL,
    AmountDue DECIMAL(18,2) NOT NULL,
    AmountPaid DECIMAL(18,2) NOT NULL DEFAULT 0,
    PaidOn DATETIME NULL,
    PaymentMethod NVARCHAR(50) NULL,
    ReferenceNo NVARCHAR(100) NULL,
    Status NVARCHAR(50) NOT NULL DEFAULT 'Pending', -- Pending, Paid, Partially Paid, Waived
    Remarks NVARCHAR(255) NULL,
    CreatedAt DATETIME NOT NULL DEFAULT GETDATE()
);

    RefundID CHAR(10) PRIMARY KEY,

    SelectedPaymentIDs NVARCHAR(MAX) NULL,     -- JSON array of PaymentIDs to delete on approval
    CreatedBy CHAR(10) NULL FOREIGN KEY REFERENCES Users(UserID),
    CreatedAt DATETIME DEFAULT GETDATE(),
    ApprovedBy CHAR(10) NULL FOREIGN KEY REFERENCES Users(UserID),
    ApprovedAt DATETIME NULL,
    Notes NVARCHAR(500) NULL
);

    WorkFlowStatus NVARCHAR(100),

    SellerCNIC NVARCHAR(200),
    SellerContact NVARCHAR(200),
    SellerAddress NVARCHAR(200),

    BuyerFatherName NVARCHAR(200),
    BuyerCNIC NVARCHAR(200),              -- Format: XXXXX-XXXXXXX-X
    BuyerPassportNo NVARCHAR(200) NULL,   -- Alphanumeric; required if no CNIC
    BuyerContact NVARCHAR(200),           -- Digits and + only
    BuyerAddress NVARCHAR(200),
     BuyerCity NVARCHAR(200),

);
